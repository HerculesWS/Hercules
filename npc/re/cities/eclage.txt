				close;

			delitem 6081,1; //Splendide_Coin

			if (Zeny < 5000) {
				mes "[Receptionist]";
				mes "It seems like you don't have enough zeny. Would you check your account again, please?";

			}
			Zeny -= 5000;
			break;

			close;
		}

		mes "Make yourself at home~";
		close2;
		percentheal 100,100;
		warp "ecl_in02",167,49;
		end;
	case 3:
		close;
	}
