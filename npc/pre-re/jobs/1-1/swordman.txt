function	script	F_JobSwdStaff	{
	mes "[Test Hall Staff]";

	next;
	if (select("Yes.", "No.") == 1) {
		mapannounce "job_sword1","Applicant " + strcharinfo(0) + " quit the test.",bc_map;
		warp "izlude_in",65,165;
		end;

	close;
}

function	script	F_JobSwdTestStaff	{
OnTouch:
	mes "[Test Hall Staff]";

	next;
	if (select("Yes.", "No.") == 1) {
		mapannounce "job_sword1","Applicant " + strcharinfo(0) + " quit the test..",bc_map;
		warp "izlude_in",65,165;
		end;
	}
	warp "job_sword1",getarg(0),getarg(1);
	end;
}
